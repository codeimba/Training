using System;

namespace c_
{
    public class Tasks
    {
        static void Main()
        {
            int n = int.Parse(Console.ReadLine());


            int currnet = 1;
            bool isBigger = false;
            for (int rows = 1; rows <= n; rows++)
            {
                for (int cols = 1; cols <= rows; cols++)
                {
                    if (currnet > n)
                    {
                        isBigger = true;
                        break;
                    }
                    System.Console.Write(currnet + " ");
                    currnet++;
                }
                if (isBigger)
                {
                    break;
                }
                System.Console.WriteLine();
            }
        }
    }
}
