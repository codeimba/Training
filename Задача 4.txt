using System;
					
public class Program
{
	public static void Main()
	{
		// входни данни от конзолата 
		int NumbPages = int.Parse(Console.ReadLine());
		int NumbPagesPerHour = int.Parse(Console.ReadLine());
		int NumbDays = int.Parse(Console.ReadLine());
		
		//изчисление за общо часове четене
		
		int HoursRed = NumbPages / NumbPagesPerHour;
		int DaysNeeded = HoursRed / NumbDays;
		
		//изходни дании за визуализация
		Console.WriteLine(DaysNeeded);
		
		
	}
}
