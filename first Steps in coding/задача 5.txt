using System;
					
public class Program
{
	public static void Main()
	{
		//цени за материали
		double PackOfPencils = 5.80;
		double PackOfMarkers = 7.20;
		double CleaningProduct = 1.20;
		
		//входни данни за броиките
		int NumbOfPencils = int.Parse(Console.ReadLine());
		int NumbOfMarkers = int.Parse(Console.ReadLine());
		int NumbOfCleanProdLitter = int.Parse(Console.ReadLine());
		int PercentDiscount = int.Parse(Console.ReadLine());
		
		float Discount = PercentDiscount / 100;
		
		//Цена на пакетите
		double PriceForPencils = PackOfPencils * NumbOfPencils;
		double PriceForMarkers = PackOfMarkers * NumbOfMarkers;
		double PriceForCleaningProduct = CleaningProduct * NumbOfCleanProdLitter;
		
		//Обща цена за материалите
		double TotalPrice = PriceForPencils + PriceForMarkers + PriceForCleaningProduct;
		
		//Цена за материалите след отстъпката
		double PriceWithDiscount = TotalPrice - (TotalPrice * Discount);
		Console.WriteLine(PriceWithDiscount);
	}
}
