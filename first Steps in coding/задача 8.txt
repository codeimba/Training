using System;
					
public class Program
{
	public static void Main()
	{
		//цена за транировки за цялата година
		int YearlySubscibtionPrice = int.Parse(Console.ReadLine());
		
		// цени за оборудване
		double PriceShoes = YearlySubscibtionPrice - (YearlySubscibtionPrice * 0.40);
		double PriceOutfit = PriceShoes - (PriceShoes * 0.20);
		double PriceBall = PriceOutfit / 4;
		double PriceAccessories = PriceBall / 5;
		
		// обща цена за екипировка
		double EquipmentPrice = PriceShoes + PriceOutfit + PriceBall + PriceAccessories + YearlySubscibtionPrice;
		float TotalPrice = (float)EquipmentPrice;
		Console.WriteLine(TotalPrice);
	}
}
