using System;
					
public class Program
{
	public static void Main()
	{
		// Входни данни от конзолата
		double DepositSum = double.Parse(Console.ReadLine());
		int MonthsDeposit = int.Parse(Console.ReadLine());
		double YearlyPercent = double.Parse(Console.ReadLine());
		
		// Схема за изчисляване
		// формула за крайна сума = депозитна сума + срок на депозита * ((депозитна сума * годишен процент) / 12)
		
		double Interest = DepositSum * (YearlyPercent / 100);
		double InterestMonthly = Interest / 12;
		double WholeSum = DepositSum + MonthsDeposit * InterestMonthly;
		
		// изходни данни 
		Console.WriteLine(WholeSum);
	}
}
