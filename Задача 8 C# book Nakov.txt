using System;
					
public class Program
{
	public static void Main()
	{	
		int a = int.Parse(Console.ReadLine());
		int b = int.Parse(Console.ReadLine());
		
		int R = 5;
		
		if ((a*a) + (b*b) == R * R) // pythagorean theorem
		{
			Console.WriteLine("true");
		}
		else
		{
			Console.WriteLine("false");
		}
	}
}
