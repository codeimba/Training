using System;
					
public class Program
{
	public static void Main()
	{
		//Входни параметри за страните на аквариума
		int length = int.Parse(Console.ReadLine());
		int width = int.Parse(Console.ReadLine());
		int height = int.Parse(Console.ReadLine());
		float Percent = float.Parse(Console.ReadLine());
		
		//обем на аквариума в кубични диаметри
		double AquariumVolume = length * width * height;
		
		//обем в литри
		double LitterVolume = AquariumVolume / 1000;
		
		//нужни литри до запълване + процент заетост на аквариума 
		double NeededLitters = LitterVolume - (LitterVolume * (Percent / 100));
		Console.WriteLine(NeededLitters);
	}
}
