using System;
					
public class Program
{
	public static void Main()
	{
		//входни данни от конзолата
		double radians = double.Parse(Console.ReadLine());
		
		// формула за изчисляване
		double degrees = radians * 180 / Math.PI;
		
		//изходни данни 
		Console.WriteLine(degrees);
	}
}
